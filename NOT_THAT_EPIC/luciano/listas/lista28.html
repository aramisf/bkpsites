5. Explique o programa abaixo:

#!/bin/bash
while true 
do
  clear
  echo " Funcoes "
  echo "---------"
  echo "[1] Data"
  echo "[2] Calendario"
  echo "[3] Sair"
  echo "--------"
  echo -n "Digite sua escolha [1-3]: "
  read escolha
  case $escolha in
     1) date; echo "Pressione qualquer tecla"; read
        ;;
     2) cal; echo "Pressione qualquer tecla"; read
        ;;
     3) echo "Fim do programa"; exit
        ;;
     *) echo "Erro! Voce deve digitar 1, 2 ou 3"
        echo "Pressione qualquer tecla"; read
        ;; 
  esac
done
